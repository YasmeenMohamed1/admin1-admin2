#!/bin/bash

while true; do
    current_time=$(date +"%H:%M")

    if [[ "$current_time" > "08:00" && "$current_time" < "17:00" ]]; then
        echo "System information at $current_time:"
        echo "------------"
        echo "Memory Usage:"
        free -h
        echo
        echo "CPU Usage:"
        top -bn1 | grep "Cpu(s)"
        echo
        echo "Disk Usage:"
        df -h
        echo
        echo "----------"
    fi

    sleep 600 # Sleep for 10 minutes (600 seconds)
done
chmod +x date.sh
